163  CLS(const std::string& msg, const Pothos::Exception& exc, int code = CODE); \
164  CLS(const CLS& exc); \

168  const char* className() const throw(); \
169  Pothos::Exception* clone() const; \
170  void rethrow() const; \
171  };
